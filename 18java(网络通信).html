计算机网络基础

一. 物理层

  1. 物理设备是传输介质, 比如双绞线/ 光纤/ wifi/ 蓝牙等
  2. 传输的数据类型: 二进制

二. 数据链路层

  1. 物理设备: 交换机
  2. 传输的数据类型: 数据帧
  3. 源和目的: 网卡的mac地址(物理地址)

三. 网络层

  1. 物理设备: 路由器
  2. 传输的数据类型: 数据包
  3. 源和目的: IP地址

四. 传输层

  1. 传输的数据类型: 数据段
  2. 源和目的: 端口号
  3. 端口范围: 0~65535, 其中 0 ~ 1024 已被系统占用
  4. 协议: TCP/UDP协议

五. 应用层

常用协议

- 基于TCP的
    HTTP 超文本传输协议
    FTP 文件传输协议
    SMTP 简单邮件传输协议
    等等

- 基于UDP的
    TFTP 普通文件

IP地址

  1. 公网地址: 任意位置都能访问的地址
  2. 私有地址: 在同一个局域网下才能访问
  3. 私有地址的分类:
    • A类私有地址: 10.0.0.0 ~ 10.255.255.255
    • B类私有地址: 172.16.00 ~ 172.31.255.255
    • C类私有地址: 192.168.0.0 ~ 192.168.255.255
  4. 本机地址: 127.0.0.1, localhost
  5. 网关地址, 在一个局域网下的路由器的地址
  6. 广播地址, 向广播地址发送数据, 该局域网下的所有客户端都能接收
  7. 子网掩码, 用来划分子网的, 二进制n个1开始, n个0结束, n个0对应的IP位全换成1, 为广播地址

传输层协议

  1. TCP 传输控制协议
  • 面向连接的
  • 可靠的
  • 传输的数据为字节流
  • 建立连接三次握手, 断开连接是四次挥手
  1. UDP 用户数据报协议
  • 没有建立连接的过程
  • 不可靠
  • 传输的数据为数据包
  • 由于没有建立连接的过程, 传输的效率会高一些

应用层协议

按照双方(客户端和服务端)约定好格式向TCP或UDP中写入数据, 这个约定好的格式就是应用层协议 HTTP 超文本传输协议 

public class HttpDemo {
    public static void main(String[] args) throws IOException {
        // 应用层协议
        Socket socket = new Socket("www.baidu.com", 80);
        OutputStream out = socket.getOutputStream();
        OutputStreamWriter w = new OutputStreamWriter(out);
        w.write("GET Https://www.baidu.com HTTP/1.1\r\n");
        w.write("Host: Https://www.baidu.com\r\n");
        w.write("\r\n");
        w.write("\r\n");    // GET请求没有请求包体
        w.flush();

        InputStream in = socket.getInputStream();
        InputStreamReader r = new InputStreamReader(in);
        BufferedReader br = new BufferedReader(r);
        StringBuilder sb = new StringBuilder();
        String line = null;

        while ((line = br.readLine()) != null) {
            sb.append(line).append("\n");
        }
        System.out.println(sb);
    }
}

Socket操作TCP协议

public class ServerMain {
    public static void main(String[] args) throws IOException {
        // Socket 套接字
        // 在java中Socket类可以直接操作TCP协议
        // 创建服务器的Socket, 设置需要监听的端口
        ServerSocket server = new ServerSocket(7890);
        // 如果监听到有客户端连接端口了
        // 就可以获取到连接过来的socket对象
        // 该方法是一个阻塞的方法
        System.out.println("服务器已启动, 开始监听7890端口");
        System.out.println("等待客户端连接...");
        while (true) {
            Socket socket = server.accept();
            // 获取到输入流, 获取客户端发送过来的数据
            BufferedReader br = new BufferedReader(new InputStreamReader(socket.getInputStream()));
            // 或取输出流, 用来给客户端发送数据
            PrintWriter p = new PrintWriter(new OutputStreamWriter(socket.getOutputStream()));

            while (true) {
                System.out.println("服务端已经连接..正在读取客户发送的内容");
                String word = br.readLine();
                p.println(word);
                p.flush();
                System.out.println("服务端回复客户端");
                if ("拜拜".equals(word)) {
                    break;
                }
            }
        }
    }
}

public class ClientMain {
    public static void main(String[] args) throws IOException {
        Scanner sc = new Scanner(System.in);
        Socket socket = new Socket("172.16.16.160", 7890);
        PrintStream p = new PrintStream(socket.getOutputStream());
        BufferedReader br = new BufferedReader(new InputStreamReader(socket.getInputStream()));
        while (true) {
            System.out.println("请输入你要说的话:");
            String s = sc.next();
            p.println(s);
            p.flush();
            String hostAddress = socket.getInetAddress().getHostAddress();   // 查看IP地址
            System.out.println(hostAddress + "客户端说:"+ s);
            if ("拜拜".equals(s)) {
                break;
            }
            String word = br.readLine();
            System.out.println("服务端说:" + word);
        }
    }
}

public class UdpServerDemo {
    public static void main(String[] args) throws IOException {
        DatagramSocket ds = new DatagramSocket(6789);
        byte[] buff = new byte[1024];
        DatagramPacket dp = new DatagramPacket(buff, buff.length);
        // 把接收到的数据存储到dp对象中
        // 阻塞的方法
        ds.receive(dp);

        // 读取数据
        byte[] data = dp.getData();
        String str = new String(data, dp.getOffset(), dp.getLength());
        System.out.println("服务端接收到数据: " + str);
    }
}

public class UdpBroadcastDemo {
    private static final String INET_BROADCAST_ADDRESS = "172.16.31.255";
    
    public static void main(String[] args) throws IOException {
        Scanner sc = new Scanner(System.in);
        DatagramSocket udp = new DatagramSocket(6789);
        new Thread(new UdpReceiver(udp)).start();

        while (true) {
            byte[] buf = sc.nextLine().getBytes();
            if (buf.length == 0) continue;
            InetAddress address = InetAddress.getByName(INET_BROADCAST_ADDRESS);
            DatagramPacket p = new DatagramPacket(buf, buf.length, address, 6789);
            udp.send(p);
        }
    }

    public static class UdpReceiver implements Runnable {
        private DatagramSocket udp;

        private UdpReceiver(DatagramSocket udp) {
            this.udp = udp;
        }

        @Override
        public void run() {
            byte[] buf = new byte[1024];
            DatagramPacket dp = new DatagramPacket(buf, buf.length);
            try {
                while (true) {
                    udp.receive(dp);
                    String msg = new String(dp.getData(), dp.getOffset(), dp.getLength());    // 去掉字节数组1024里空的地方
                    String address = dp.getAddress().getHostAddress();
                    if (msg.isEmpty() || msg.trim().isEmpty()) continue;
                    System.out.println(address + ": " + msg);
                }
            } catch (IOException e) {
                e.printStackTrace();
            }
        }
    }
}

public class HttpDemo {
    public static void main(String[] args) throws IOException {
        String url = "https://ss1.bdstatic.com/70cFuXSh_Q1YnxGkpoWK1HF6hhy/it/u=363149310,1133088467&fm=15&gp=0.jpg";
        // OkHttp 发送Http请求的框架
        /*
        发送Http请求步骤:
        1. 请求的地址
        2. 指定请求方法(GET/POST)
            2.1 GET: 一般用来向服务器请求数据
            2.2 POST: 一般用来向服务器发送数据
        3. 设置请求头    简单的请求可以省略
        4. 设置请求体    GET请求没有
        5. 发送请求
        6. 处理响应
         */
        OkHttpClient client = new OkHttpClient.Builder().build();
        Request request = new Request.Builder()
                .url(url)   // 1.
                .get()  // 2.
                //.head() // 3.
                .build();
        Response resp = client.newCall(request).execute(); // 5.
        InputStream in = resp.body().byteStream(); // 6.
        // 查看文件总大小
        long size = resp.body().contentLength();
        System.out.println(size);
        byte[] buff = new byte[1024];
        // 查看进度
        double currentSize = 0;
        FileOutputStream out = new FileOutputStream("girl.jpg");
        int length = -1;
        while ((length = in.read(buff)) != -1) {
            out.write(buff, 0 , length);
            currentSize += length;
            System.out.println("进度:" + (currentSize / size * 100) + "%");
        }
        out.close();
        in.close();
        // 断点续传
        RandomAccessFile file = new RandomAccessFile("test.jpg", "rw");
        file.seek(2000L);
    }
}
