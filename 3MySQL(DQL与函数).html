函数

between A and B 包含A和B, A必须比B小

is null

ifnull(A, 0) 如果A为null, 则把A临时变成0

case operator

select *, case 
when score between 80 and 100 then '优秀'
when score between 60 and 79 then '及格'
else '不及格' end as '成绩分类'
from t_score;

select *,
if(score >= 90, '录取', '淘汰') as '状态'
from t_score order by score desc;   // order by不能直接接到where后面

ifnull(列名, 值) 如果列中的内容为空, 变为值

round(小数, n) 保留n位小数, 返回四舍五入后的值

bin(n) 把n转成2进制

oct(n) 把n转成8进制

hex(n) 把n转成16进制

concat(A, 2) 拼接字符串

sum(分数) 求总分数

sum(分数 > 60) 求分数大于60 的分数个数, true就+1

sum(分数 > 60 and 分数 < 100) 必须加and

group_concat(distinct expr order by expr separator ' ') 分组拼接

substring(str,pos), substring(str from pos), substring(str,pos,len), substring(str from pos for len)

# 1. 查询"1"课程比"2"课程成绩高的学生的信息及课程分数
select t1.stu_id, stu_name, birthday, stu_gender,
s1.score as '课程1的分数', s2.score as '课程2的分数'
from
t_students t1, t_score s1, t_score s2
where
t1.stu_id = s1.stu_id and 
t1.stu_id = s2.stu_id and 
s1.course_id = 1 and
s2.course_id = 2 and 
s1.score > s2.score;

# 2. 查询"语文"课程比"数学"课程成绩高的学生的信息及课程分数
select st.stu_id, stu_name, birthday, stu_gender,
s1.score as '语文的分数', s2.score as '数学的分数'
from
t_students st, t_score s1, t_score s2, t_courses t1, t_courses t2
where
s1.course_id = t1.course_id and
s2.course_id = t2.course_id and
t1.course_name = '语文' and t2.course_name = '数学' and
st.stu_id = s1.stu_id and
st.stu_id = s2.stu_id and
s1.score > s2.score;

# 查询所有学生的选课数量和总成绩大于100分的学生信息, 并按照总成绩倒序排序
select 
st.stu_name, count(course_id), sum(s1.score) as score_sum
from
t_students st, t_score s1
where 
st.stu_id = s1.stu_id 
group by stu_name
having sum(s1.score) > 100
order by sum(s1.score) desc;

from t_students s

left join t_courses cou on sc.course_id = cou.course_id 
group by s.stu_id order by sum(sc.score) desc;

# 查询两门及其以上不及格课程的同学的学号，姓名及其平均成绩
select st.stu_name, avg(ifnull(s1.score, 0)) as '平均数'
from (t_students st, t_courses c1) left join t_score s1
on st.stu_id = s1.stu_id and c1.course_id = s1.course_id
group by st.stu_id
having group_concat(ifnull(s1.score, 0)) < 60
and count(ifnull(s1.score, 0) < 60) >= 2;

# 按学号查看各科成绩
select stu.stu_id,
stu.stu_name,
max(if(crs.course_id = 1, sc.score, 0)) as '语文成绩',
max(case when crs.course_id = 2 then ifnull(sc.score, 0) end) as '数学成绩',
max(case when crs.course_id = 3 then ifnull(sc.score, 0) end) as '英语成绩'
from t_students stu
cross join t_courses crs
left join t_score sc
on stu.stu_id = sc.stu_id and 
crs.course_id = sc.course_id
group by stu.stu_id;

# 对语文成绩进行排名
set @row=0,@rank=0,@score=null; 
select @row := @row + 1 '编号', temp.stu_id, temp.stu_name,
@rank := if(@score=temp.score,@rank,@row) as '排名',
@score := temp.score as '语文成绩'
from(
	select st.stu_id, st.stu_name, ifnull(s1.score, 0) as 'score'
	from t_students st 
	left join t_score s1 
	on st.stu_id = s1.stu_id 
	and s1.course_id = 1 
	order by s1.score desc
) temp;

注意:

  • CURRENT_DATE 获取当前日期
  • distinct 去重
  • group by 前面不能有and
  • count()满足的条件不能在where里, 要写在having里
